### 第六步：配置全局导航守卫（路由权限控制）

#### 6.1 需求说明

* **购物车页面**（`/cart`）：需要登录才能访问

* **个人中心页面**（`/profile`）：需要登录才能访问

* **首页**（`/`）：公开访问

* **登录页**（`/login`）：公开访问

#### 6.2 修改路由配置，添加 meta 字段

修改 `src/router/index.js`，给需要登录的页面添加 `meta.requiresAuth`：

```javascript
// src/router/index.js
import { createRouter, createWebHistory } from 'vue-router'
import HomeView from '../views/HomeView.vue'
import LoginView from '../views/LoginView.vue'
import CartView from '../views/CartView.vue'
import ProfileView from '../views/ProfileView.vue'

const routes = [
  {
    path: '/',
    name: 'home',
    component: HomeView
  },
  {
    path: '/login',
    name: 'login',
    component: LoginView
  },
  {
    path: '/cart',
    name: 'cart',
    component: CartView,
    meta: { requiresAuth: true }  // 👈 需要登录
  },
  {
    path: '/profile',
    name: 'profile',
    component: ProfileView,
    meta: { requiresAuth: true }  // 👈 需要登录
  }
]

const router = createRouter({
  history: createWebHistory(import.meta.env.BASE_URL),
  routes
})

export default router
```


#### 6.3 添加全局导航守卫

在 `src/router/index.js` 中，在 `router` 导出之前添加守卫代码：

```javascript
// src/router/index.js（完整版）
import { createRouter, createWebHistory } from 'vue-router'
import HomeView from '../views/HomeView.vue'
import LoginView from '../views/LoginView.vue'
import CartView from '../views/CartView.vue'
import ProfileView from '../views/ProfileView.vue'

const routes = [
  {
    path: '/',
    name: 'home',
    component: HomeView
  },
  {
    path: '/login',
    name: 'login',
    component: LoginView
  },
  {
    path: '/cart',
    name: 'cart',
    component: CartView,
    meta: { requiresAuth: true }
  },
  {
    path: '/profile',
    name: 'profile',
    component: ProfileView,
    meta: { requiresAuth: true }
  }
]

const router = createRouter({
  history: createWebHistory(import.meta.env.BASE_URL),
  routes
})

// ===== 全局前置导航守卫 =====
router.beforeEach((to, from, next) => {
  // 判断目标页面是否需要登录
  if (to.meta.requiresAuth) {
    // 检查是否已登录（从 localStorage 读取 token）
    const token = localStorage.getItem('token')
    
    if (token) {
      // 已登录，允许访问
      next()
    } else {
      // 未登录，跳转到登录页，并携带重定向地址
      next({
        path: '/login',
        query: { redirect: to.fullPath }
      })
    }
  } else {
    // 不需要登录，直接放行
    next()
  }
})

export default router
```

```javascript
// src/store/userStore.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useUserStore = defineStore('user', () => {
  // ===== State（状态） =====
  // 从 localStorage 读取 token，实现页面刷新后登录状态保持
  const token = ref(localStorage.getItem('token') || '')
  const user = ref(JSON.parse(localStorage.getItem('userInfo') || 'null'))

  // ===== Getters（计算属性） =====
  // 判断是否已登录
  const isLoggedIn = computed(() => {
    return !!token.value && !!user.value
  })

  // 获取用户姓名
  const userName = computed(() => {
    return user.value?.name || user.value?.username || '用户'
  })

  // 获取用户头像
  const userAvatar = computed(() => {
    return user.value?.avatar || 'https://api.dicebear.com/7.x/avataaars/svg?seed=default'
  })

  // ===== Actions（方法） =====
  // 登录方法
  function login(loginData) {
    // 暂时使用模拟数据，第三部分会对接真实 API
    // 模拟登录验证
    if (loginData.username === 'admin' && loginData.password === '123456') {
      const mockUser = {
        id: 1,
        username: 'admin',
        name: '管理员',
        email: 'admin@example.com',
        role: 'admin',
        avatar: 'https://api.dicebear.com/7.x/avataaars/svg?seed=admin'
      }
      const mockToken = 'mock_token_1_123456'

      // 保存到 state
      token.value = mockToken
      user.value = mockUser

      // 保存到 localStorage（持久化）
      localStorage.setItem('token', mockToken)
      localStorage.setItem('userInfo', JSON.stringify(mockUser))

      return { success: true, data: mockUser }
    }

    // 测试账号：user / 123456
    if (loginData.username === 'user' && loginData.password === '123456') {
      const mockUser = {
        id: 2,
        username: 'user',
        name: '普通用户',
        email: 'user@example.com',
        role: 'user',
        avatar: 'https://api.dicebear.com/7.x/avataaars/svg?seed=user'
      }
      const mockToken = 'mock_token_2_123456'

      token.value = mockToken
      user.value = mockUser
      localStorage.setItem('token', mockToken)
      localStorage.setItem('userInfo', JSON.stringify(mockUser))

      return { success: true, data: mockUser }
    }

    // 登录失败
    return { success: false, message: '用户名或密码错误' }
  }

  // 退出登录
  function logout() {
    token.value = ''
    user.value = null
    localStorage.removeItem('token')
    localStorage.removeItem('userInfo')
  }

  // 初始化登录状态（页面加载时从 localStorage 恢复）
  function initAuth() {
    const savedToken = localStorage.getItem('token')
    const savedUser = localStorage.getItem('userInfo')
    
    if (savedToken && savedUser) {
      token.value = savedToken
      user.value = JSON.parse(savedUser)
    }
  }

  return {
    // State
    token,
    user,
    // Getters
    isLoggedIn,
    userName,
    userAvatar,
    // Actions
    login,
    logout,
    initAuth
  }
})
```

### 第三步：创建购物车 Store（cartStore）

#### 3.1 创建 cartStore.js

创建 `src/store/cartStore.js`：

```javascript
// src/store/cartStore.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useCartStore = defineStore('cart', () => {
  // ===== State =====
  // 购物车商品列表，每个商品包含：id, name, price, image, quantity
  const items = ref([])

  // ===== Getters（计算属性） =====
  // 购物车总数量
  const totalCount = computed(() => {
    return items.value.reduce((sum, item) => sum + item.quantity, 0)
  })

  // 购物车总价
  const totalPrice = computed(() => {
    return items.value.reduce((sum, item) => sum + item.price * item.quantity, 0)
  })

  // 判断购物车是否为空
  const isEmpty = computed(() => {
    return items.value.length === 0
  })

  // 根据商品ID查找购物车中的商品
  const findItem = (productId) => {
    return items.value.find(item => item.id === productId)
  }

  // ===== Actions =====
  // 添加商品到购物车
  function addItem(product, quantity = 1) {
    const existingItem = findItem(product.id)
    
    if (existingItem) {
      // 如果商品已在购物车中，增加数量
      existingItem.quantity += quantity
    } else {
      // 否则添加新商品
      items.value.push({
        id: product.id,
        name: product.name,
        price: product.price,
        image: product.image || 'https://picsum.photos/200/200?random=' + product.id,
        quantity: quantity
      })
    }
  }

  // 从购物车移除商品
  function removeItem(productId) {
    const index = items.value.findIndex(item => item.id === productId)
    if (index !== -1) {
      items.value.splice(index, 1)
    }
  }

  // 更新商品数量
  function updateQuantity(productId, quantity) {
    const item = findItem(productId)
    if (item) {
      if (quantity <= 0) {
        // 数量为0或负数时移除商品
        removeItem(productId)
      } else {
        item.quantity = quantity
      }
    }
  }

  // 清空购物车
  function clearCart() {
    items.value = []
  }

  // 从 localStorage 加载购物车数据（持久化）
  function loadCart() {
    const saved = localStorage.getItem('cartItems')
    if (saved) {
      try {
        items.value = JSON.parse(saved)
      } catch (e) {
        items.value = []
      }
    }
  }

  // 保存购物车数据到 localStorage
  function saveCart() {
    localStorage.setItem('cartItems', JSON.stringify(items.value))
  }

  return {
    // State
    items,
    // Getters
    totalCount,
    totalPrice,
    isEmpty,
    findItem,
    // Actions
    addItem,
    removeItem,
    updateQuantity,
    clearCart,
    loadCart,
    saveCart
  }
})
```

#### 12.2 在购物车变化时自动保存

修改 `src/store/cartStore.js`，在每次修改时自动保存：

```javascript
// src/store/cartStore.js（关键修改）
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useCartStore = defineStore('cart', () => {
  const items = ref([])

  const totalCount = computed(() => {
    return items.value.reduce((sum, item) => sum + item.quantity, 0)
  })

  const totalPrice = computed(() => {
    return items.value.reduce((sum, item) => sum + item.price * item.quantity, 0)
  })

  const isEmpty = computed(() => {
    return items.value.length === 0
  })

  const findItem = (productId) => {
    return items.value.find(item => item.id === productId)
  }

  // 👇 保存函数
  function saveToStorage() {
    localStorage.setItem('cartItems', JSON.stringify(items.value))
  }

  function addItem(product, quantity = 1) {
    const existingItem = findItem(product.id)
    if (existingItem) {
      existingItem.quantity += quantity
    } else {
      items.value.push({
        id: product.id,
        name: product.name,
        price: product.price,
        image: product.image || 'https://picsum.photos/200/200?random=' + product.id,
        quantity: quantity
      })
    }
    saveToStorage()  // 👈 自动保存
  }

  function removeItem(productId) {
    const index = items.value.findIndex(item => item.id === productId)
    if (index !== -1) {
      items.value.splice(index, 1)
      saveToStorage()
    }
  }

  function updateQuantity(productId, quantity) {
    const item = findItem(productId)
    if (item) {
      if (quantity <= 0) {
        removeItem(productId)
      } else {
        item.quantity = quantity
        saveToStorage()
      }
    }
  }

  function clearCart() {
    items.value = []
    saveToStorage()
  }

  function loadCart() {
    const saved = localStorage.getItem('cartItems')
    if (saved) {
      try {
        items.value = JSON.parse(saved)
      } catch (e) {
        items.value = []
      }
    }
  }

  return {
    items,
    totalCount,
    totalPrice,
    isEmpty,
    findItem,
    addItem,
    removeItem,
    updateQuantity,
    clearCart,
    loadCart
  }
})
```

### ❓ 常见问题

**Q1：登录后页面没有自动跳转？**

* 检查 `LoginView.vue` 中的 `router.push(redirect)` 是否正确

* 检查 `route.query.redirect` 是否获取到

**Q2：购物车数据刷新后丢失？**

* 检查 `cartStore.saveToStorage()` 是否在每次修改时调用

* 检查 `main.js` 中是否调用了 `cartStore.loadCart()`

**Q3：导航守卫不生效？**

* 检查路由的 `meta.requiresAuth` 是否设置为 `true`

* 检查 `localStorage` 中是否有 `token`

**Q4：商品详情页 404？**

* 检查路由配置中是否添加了 `path: '/product/:id'`

* 检查 `router-link` 的 `to` 路径是否正确

```javascript
// 简单模拟 JWT token（第三部分先用模拟，后续可升级为真实 JWT）

function generateToken(user) {
  // 格式: mock_token_用户ID_时间戳
  return `mock_token_${user.id}_${Date.now()}`
}

function verifyToken(token) {
  if (!token || !token.startsWith('mock_token_')) {
    return null
  }
  
  try {
    const parts = token.split('_')
    if (parts.length !== 3) return null
    const userId = parseInt(parts[1])
    if (isNaN(userId)) return null
    return { userId }
  } catch (error) {
    return null
  }
}

module.exports = { generateToken, verifyToken }
```

```javascript
// src/api/request.js
import axios from 'axios'

// ===== 创建 Axios 实例 =====
const request = axios.create({
  baseURL: 'http://localhost:3000',  // 后端地址
  timeout: 10000,                    // 请求超时时间
  headers: {
    'Content-Type': 'application/json'
  }
})

// ===== 请求拦截器 =====
request.interceptors.request.use(
  (config) => {
    // 在发送请求之前做些什么
    const token = localStorage.getItem('token')
    
    if (token) {
      // 将 Token 添加到请求头
      config.headers.Authorization = `Bearer ${token}`
    }
    
    return config
  },
  (error) => {
    // 对请求错误做些什么
    console.error('请求错误:', error)
    return Promise.reject(error)
  }
)

// ===== 响应拦截器 =====
request.interceptors.response.use(
  (response) => {
    // 对响应数据做点什么
    const res = response.data
    
    // 后端统一返回格式: { code, message, data }
    if (res.code === 200) {
      return res  // 成功，直接返回
    } else {
      // 业务错误（如 401 未登录、404 资源不存在等）
      return Promise.reject(res)
    }
  },
  (error) => {
    // 对响应错误做点什么（网络错误、超时等）
    if (error.response) {
      // 服务器返回了错误状态码
      const { status, data } = error.response
      
      // 401 未认证（Token 过期或无效）
      if (status === 401) {
        // 清除本地存储
        localStorage.removeItem('token')
        localStorage.removeItem('userInfo')
        
        // 跳转到登录页
        window.location.href = '/login'
      }
      
      console.error('API 错误:', data?.message || '请求失败')
      return Promise.reject(data || { message: '请求失败' })
    } else if (error.request) {
      // 请求已发出但没有收到响应
      console.error('网络错误: 服务器未响应')
      return Promise.reject({ message: '网络错误，请检查后端是否运行' })
    } else {
      // 其他错误
      console.error('请求配置错误:', error.message)
      return Promise.reject({ message: error.message })
    }
  }
)

export default request
```

```javascript
        // 保存到 state
        token.value = newToken
        user.value = userInfo
        
        // 保存到 localStorage
        localStorage.setItem('token', newToken)
        localStorage.setItem('userInfo', JSON.stringify(userInfo))
        
        return { success: true, data: userInfo }

        localStorage.setItem('userInfo', JSON.stringify(userInfo))
```

```javascript
import { ref } from 'vue'
import { productApi } from '../api/productApi'

export const useProductStore = defineStore('product', () => {
  // ===== State =====
  const products = ref([])
  const loading = ref(false)
  const error = ref(null)

  // ===== Actions =====
  
  // 获取所有商品
  async function fetchProducts() {
    loading.value = true
    error.value = null
    
    try {
      const response = await productApi.getProducts()
      
      if (response.code === 200) {
        products.value = response.data
        return { success: true, data: response.data }
      }
      
      error.value = response.message || '获取商品列表失败'
      return { success: false, message: error.value }
    } catch (err) {
      error.value = err.message || '网络错误'
      return { success: false, message: error.value }
    } finally {
      loading.value = false
    }
  }

  // 获取单个商品
  async function fetchProductById(id) {
    loading.value = true
    error.value = null
    
    try {
      const response = await productApi.getProductById(id)
      
      if (response.code === 200) {
        return { success: true, data: response.data }
      }
      
      error.value = response.message || '获取商品详情失败'
      return { success: false, message: error.value }
    } catch (err) {
      error.value = err.message || '网络错误'
      return { success: false, message: error.value }
    } finally {
      loading.value = false
    }
  }

  return {
    products,
    loading,
    error,
    fetchProducts,
    fetchProductById
  }
```

```vue
        </p>
```

```vue
        <div class="quantity-selector">
          <button @click="decreaseQuantity" :disabled="quantity <= 1">−</button>
          <span>{{ quantity }}</span>
          <button @click="increaseQuantity" :disabled="quantity >= product.stock">+</button>

<script setup>
import { ref, onMounted } from 'vue'
import { useRoute, useRouter } from 'vue-router'
import { useProductStore } from '../store/productStore'
import { useCartStore } from '../store/cartStore'

const route = useRoute()
const router = useRouter()
const productStore = useProductStore()
const cartStore = useCartStore()

// 数据
const product = ref(null)
const quantity = ref(1)
const message = ref('')
const loading = ref(false)
const error = ref('')

// 加载商品详情
const loadProduct = async () => {
  const productId = parseInt(route.params.id)
  
  loading.value = true
  error.value = ''
  
  try {
    const result = await productStore.fetchProductById(productId)
    
    if (result.success) {
      product.value = result.data
    } else {
      error.value = result.message || '加载失败'
      product.value = null
    }
  } catch (err) {
    error.value = '网络错误，请稍后重试'
    product.value = null
  } finally {
    loading.value = false
  }
}

// 数量控制
const decreaseQuantity = () => {
  if (quantity.value > 1) quantity.value--
}

const increaseQuantity = () => {
  if (product.value && quantity.value < product.value.stock) {
    quantity.value++
  }
}

// 加入购物车
const addToCart = () => {
  if (!product.value) return
  
  cartStore.addItem(product.value, quantity.value)
  cartStore.saveCart()
  
  message.value = `✅ 已添加 ${quantity.value} 件 "${product.value.name}" 到购物车`
  
  setTimeout(() => {
    message.value = ''
  }, 3000)
}

// 组件挂载时加载数据
onMounted(() => {
  loadProduct()
})
</script>
```

## ❓ 常见问题

**Q1：后端启动报错 "address already in use"**

* 端口 3000 被占用，修改 `server.js` 中的 `PORT` 为其他端口（如 3001）

* 同时修改前端 `request.js` 中的 `baseURL`

**Q2：前端请求报 CORS 错误**

* 确认后端 CORS 配置正确

* 确认后端已启动

* 确认 `baseURL` 指向正确的后端地址

**Q3：登录后 token 无效**

* 检查 `localStorage` 中是否保存了 token

* 检查请求拦截器是否正确添加了 `Authorization` 头

**Q4：商品列表为空**

* 确认后端 `/api/products` 接口返回数据

* 检查浏览器 Network 面板查看请求状态
